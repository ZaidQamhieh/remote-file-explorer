```dart
// app/lib/core/storage/cache_manager.dart
import 'dart:io';
import 'package:path_provider/path_provider.dart';

class CacheManager {
  /// Sum of all file sizes in [dirs], in bytes.
  static Future<int> totalSizeOf(List<Directory> dirs) async {
    var total = 0;
    for (final dir in dirs) {
      if (!dir.existsSync()) continue;
      await for (final entity in dir.list(recursive: true)) {
        if (entity is File) {
          try {
            total += await entity.length();
          } catch (_) {}
        }
      }
    }
    return total;
  }

  /// Delete all files in [dirs]. Returns bytes freed.
  static Future<int> clearDirs(List<Directory> dirs) async {
    var freed = 0;
    for (final dir in dirs) {
      if (!dir.existsSync()) continue;
      await for (final entity in dir.list()) {
        try {
          final stat = await entity.stat();
          if (entity is File) {
            freed += stat.size;
            await entity.delete();
          } else if (entity is Directory) {
            freed += await totalSizeOf([entity]);
            await entity.delete(recursive: true);
          }
        } catch (_) {}
      }
    }
    return freed;
  }

  /// The app's cache directories (temp + app cache).
  static Future<List<Directory>> appCacheDirs() async {
    return [
      await getTemporaryDirectory(),
      await getApplicationCacheDirectory(),
    ];
  }

  /// Convenience: total size of all app cache dirs.
  static Future<int> appCacheSize() async =>
      totalSizeOf(await appCacheDirs());

  /// Convenience: clear all app cache dirs.
  static Future<int> clearAppCache() async =>
      clearDirs(await appCacheDirs());
}
```

```dart
// app/lib/core/security/biometric_lock.dart
import 'package:local_auth/local_auth.dart';

class BiometricLock {
  BiometricLock._();
  static final _auth = LocalAuthentication();

  static Future<bool> isAvailable() async {
    try {
      final canCheck = await _auth.canCheckBiometrics;
      final isSupported = await _auth.isDeviceSupported();
      return canCheck || isSupported;
    } catch (_) {
      return false;
    }
  }

  static Future<bool> authenticate({String reason = 'Unlock app'}) async {
    try {
      return await _auth.authenticate(
        localizedReason: reason,
        options: const AuthenticationOptions(
          stickyAuth: true,
          biometricOnly: false,
        ),
      );
    } catch (_) {
      return false;
    }
  }
}
```

```dart
import 'package:flutter/material.dart';

import '../l10n_ext.dart';
import 'biometric_lock.dart';

class LockController extends ValueNotifier<bool> {
  LockController() : super(false);

  void lock() => value = true;

  Future<void> unlock() async {
    if (!value) return;
    final ok = await BiometricLock.authenticate();
    if (ok) value = false;
  }
}

class LockOverlay extends StatelessWidget {
  const LockOverlay({super.key, required this.controller});

  final LockController controller;

  @override
  Widget build(BuildContext context) {
    return ValueListenableBuilder<bool>(
      valueListenable: controller,
      builder: (context, locked, _) {
        if (!locked) return const SizedBox.shrink();
        final scheme = Theme.of(context).colorScheme;
        return Material(
          color: scheme.surface,
          child: Center(
            child: Column(
              mainAxisSize: MainAxisSize.min,

                Icon(Icons.lock_outline, size: 64, color: scheme.primary),
```

```go
func TestDownloadHandler(t *testing.T) {
    dir := t.TempDir()
    // Seed a test file
    os.WriteFile(filepath.Join(dir, "test.txt"), []byte("hello"), 0644)
    ops := fsops.New(dir)

    handler := downloadHandler(ops, /* settings */)

    tests := []struct {
        name   string
        query  string
        status int
    }{
        {"valid file", "?path=/test.txt", 200},
        {"missing file", "?path=/nope.txt", 404},
        {"no path param", "", 400},
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            req := httptest.NewRequest("GET", "/v1/content"+tt.query, nil)
            rec := httptest.NewRecorder()
            handler.ServeHTTP(rec, req)
            if rec.Code != tt.status {
                t.Errorf("got %d, want %d", rec.Code, tt.status)
            }
        })
    }
}
```
